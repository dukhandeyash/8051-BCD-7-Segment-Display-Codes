CHAR_S EQU #92H
CHAR_A EQU #88H
CHAR_L EQU #C7H
CHAR_E EQU #86H

BIG_DELAY EQU 5000 
LOOP_COUNT EQU 10

REPEAT:

MOV R6, #0

LOOP_CHAR:
  MOV P3, #0
  MOV P1, CHAR_S + R6
  ACALL DELAY

  MOV P3, #1
  MOV P1, CHAR_S + R6
  ACALL DELAY

  MOV P3, #2
  MOV P1, CHAR_S + R6
  ACALL DELAY

  MOV P3, #3
  MOV P1, CHAR_S + R6
  ACALL DELAY

  INC R6
  CJNE R6, LOOP_COUNT, LOOP_CHAR

MOV R7, BIG_DELAY

LOOP_DELAY:
  DJNZ R7, LOOP_DELAY

SJMP REPEAT

DELAY:
	MOV TL0, #0FH
	MOV TH0, #0FEH
	MOV TMOD, #01H
	MOV TCON, #10H
